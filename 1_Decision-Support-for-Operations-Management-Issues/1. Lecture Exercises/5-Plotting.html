#cost function
cost_eoq_fun <- function(q, d, co, cl) {
  # returns total cost per period
  # d...demand
  # q...lot size
  # co...ordering cost
  # cl...stock holding cost
  ((1/2)*cl*q)+((d/q)*co)
}

#economic ordering quantity function
eoq_fun <- function(co, d, cl) {
  # return optimal lot size
  # d...demand
  # co...ordering cost
  # cl...stock holding cost
  sqrt((2*co*d)/cl)
}

Consider the in the joint economic lot size model. Supplier and customer face setup costs \((c_{set}^{sup},c_{set}^{cust})\) as well as stock-holding cost rates \((c_{sh}^{sup},c_{sh}^{cust})\) The demand rate of the customers is and production rate of the supplier is (with < ). The cost functions of supplier and customer are: \(C^{cust}(q) =\frac{}{q}\times c_{set}^{cust} + c_{sh}^{cust}\times \frac{q}{2}\) and \(C^{supp}(q) =\frac{}{q}\times c_{set}^{sup} + c_{sh}^{sup}\times \frac{q}{2}\times \frac{}{}\) Plot the cost functions of supplier, customer, and the SC (i.e., total cost function) in different colors. Add a legend to the plot.

#which.min= index of the first minimum of numeric or logical vector

jel_func <- function(q, c.supp.set, c.supp.sh, c.cust.set, c.cust.sh, lambda, mu){
  #costs in joint exonomic lot sizing model
  if(lambda > mu) stop("infeasible(lambda > mu)")#check feasibility
  c.cust <- lambda/q * c.cust.set + c.cust.sh * q/2
  c.supp <-  lambda/q * c.supp.set  + c.supp.sh * q/2 * lambda/mu
  c.sc <- c.supp + c.cust
  return(data.frame(cost.customer = c.cust, cost.supplier = c.supp, cost.sc = c.sc))
}

Exercise 6:

Consider the contract design problem with buy-back option (Thonemann, 2010, p.479). Given are unit cost per product unit \(c\) of the supplier, the sales price of the supplier \(w\), the sales price of retailer \(r\), the scrap revenue \(v\) of the SC as well as normally distributed demand with \(y\)~\(N(\mu,\sigma^2)\). To be determined is the buy-back price \(b\) such that the optimal profit of the SC is generated. The optimal SC profit is obtained if the retailer orders \(q=\mu +\sigma \times \Phi^{-1} (CR)\) \(CR= Critical Ratio\)units, whereby \(CR = \frac{r-c}{r-v}\) is the SCs critical ratio. For the optimizing buy-back rice \(b\) holds: \(b(w)=-r\times \frac{c-v}{r-c}+w\times \frac{r-v}{r-c}\)

The profits of retailer and supplier in case of optimal combinations of \(b\) and \(w\) holds \(\\\) \(G^{ret.}(w) = (r-w)\times \mu-(r-b)\times \sigma \times \Phi^{-1} (CR)\\\) and \(\\\) \(G^{sup.}(w) = (w-c)\times \mu-(b-v)\times \sigma \times \Phi^{-1} (CR)\\\) Plot the profit functions of retailer and supplier as well as the buy-back price function depending on \(w \\\).

\(c= unit \space Cost \space per \space product \space unit \space of \space supplier \\\) \(w=\)wholesale or supplier price\(\\\) \(r=\)retailers sales price \(\\\) \(v=\) scrap revenue of SC(supplier and customer)\(\\\) \(b=\) buy back Price\(\\\) \(CR=\)critical ratio SC \(\\\) \(G^{ret.}(w)=\) retailer or customer profit given \(b\) and \(w \\\) \(G^{sup.}(w)=\) supplier or wholesaler profit given \(b\) and \(w \\\) #buy back function contains = {buy back price,profit of supplier, profit of customer, profit of supplier and customer}

buy_back_func <- function(w, r, c, v, mu, sigma){
  # costs & buy-back price in buy-back model
  crit.rat.sc <- (r-c)/(r-v) # critical ratio SC
  z.sc <- qnorm(crit.rat.sc)#Density distrib func and quantile function and random generation. 
  b <- -(c-v)/(r-c)*r + (r-v)/(r-c)*w
  profit.cust<- (r-w)*mu - (r-b)*dnorm(z.sc)* sigma
  profit.supp <- (w-c)*mu - (b-v)*dnorm(z.sc)* sigma
  return(data.frame(buy.back.price = b, profit.supplier = profit.supp, profit.cust = profit.cust, profit.sc = profit.cust+profit.supp ))
}
